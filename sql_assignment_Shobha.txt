PROCESS:

SELECT s.name as stage_name,p.name as process_name,t.name as task_name ,c.name  as checklist_name
from process p join process_stage_mapping psm on  p.id=psm.process_id 
join stage s on s.ID =psm.stage_id
join stage_task_mapping stm on s.ID =stm.stage_id 
JOIN task t on stm.task_id =t.ID 
join task_checklist_mapping tcm on tcm.task_id =t.ID 
join checklist c on c.ID =tcm.checklist_id 
where p.ID=1;
